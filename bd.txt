create table role (
	id serial primary key,
	title VARCHAR(50)
);
create table addres_coffee (
	id serial primary key,
	text VARCHAR(200)
);
create table status (
	id serial primary key,
	title VARCHAR(50)
);
create table user_account (
	id serial primary key,
	surname VARCHAR(50),
	name VARCHAR(50),
	patronymic VARCHAR(50),
	email VARCHAR(255),
	phone_number VARCHAR(11),
	login VARCHAR(30),
	password text,
	role int,
	status int,
	foreign key(role) references role(id) on delete cascade,
	foreign key(status) references status(id) on delete cascade
);
create table user_work_addresses (
    id serial primary key,
    user_id int not null,
    address int not null,
    is_primary boolean default false,
    start_date date,
    end_date date,
    foreign key(user_id) references user_account(id) on delete cascade,
    foreign key(address) references addres_coffee(id) on delete cascade
);
CREATE TABLE medical_exams (
    id serial primary key,
    user_id INTEGER NOT NULL,
    exam_date DATE NOT NULL,
    doctor_organization VARCHAR(200),
    doctor_name VARCHAR(150),
    conclusion VARCHAR(50) NOT NULL,
    foreign key(user_id) references user_account(id) on delete cascade
);
CREATE TABLE certifications (
    id serial primary key,
    user_id INTEGER NOT NULL,
    cert_type VARCHAR(100) NOT NULL,
    issue_date DATE NOT NULL,
    protocol_number VARCHAR(50),
    result VARCHAR(20) NOT NULL,
    foreign key(user_id) references user_account(id) on delete cascade
);
create table schedule (
    id serial primary key,
    user_address_id int not null,
    week_day int not null,
    start_time time not null,
    end_time time not null,
	valid_from date default CURRENT_DATE,
    valid_to date default Null,
    foreign key(user_address_id) references user_work_addresses(id) on delete cascade
);
